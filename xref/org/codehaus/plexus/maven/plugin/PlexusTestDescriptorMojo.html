25  import java.io.File;
26  import java.util.Collections;
27  
28  import org.apache.maven.plugin.MojoExecutionException;
29  import org.apache.maven.artifact.handler.ArtifactHandler;
30  
31  /**
32   * Generates a Plexus <tt>components.xml</tt> component descriptor file from test source (javadoc) or test class annotations.
33   *
34   * @goal test-descriptor
35   * @phase process-test-classes
36   * @requiresDependencyResolution test
37   *
38   * @author <a href="mailto:trygvis@inamo.no">Trygve Laugst&oslash;l</a>
39   * @version $Id$
40   * @since 1.3.4
41   */
42  public class PlexusTestDescriptorMojo
43      extends AbstractDescriptorMojo
44  {
45      //
46      // FIXME: When running as process-classes, which is required to deal with the class-based annotation gleaning
47      //        we can't put the generated file into resources, as it will never make it to the correct directory
48      //
49      //        expression="${project.build.directory}/generated-test-resources/plexus/"
50      //
51  
52      /**
53       * The output directory where the descriptor is written.
54       *
55       * @parameter expression="${project.build.testOutputDirectory}"
56       * @required
57       */
58      private File outputDirectory;
59  
60      public void execute() throws MojoExecutionException {
61          // Only execute if the current project looks like its got Java bits in it
62          ArtifactHandler artifactHandler = getMavenProject().getArtifact().getArtifactHandler();
63  
64          if (!"java".equals(artifactHandler.getLanguage())) {
65              log.debug("Not executing on non-Java project");
66          }
67          else {
68              generateDescriptor(TEST_SCOPE, new File(outputDirectory, fileName));
69  
70              getMavenProjectHelper().addTestResource( getMavenProject(),
71                                                       outputDirectory.getAbsolutePath(),
72                                                       Collections.EMPTY_LIST,
73                                                       Collections.EMPTY_LIST );
74          }
75      }
76  }
